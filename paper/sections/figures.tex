% --- Fig 1: Spectra + Identifiability Combined ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_spectra_combined.pdf}
\capt[Diffusivity Spectra By Tissue Type With MAP And NUTS Comparison.]{%
Box plots of spectral fractions for normal (top) and tumor (bottom) ROIs in
peripheral zone (left) and transition zone (right), comparing MAP (green) and
NUTS (orange) estimates.
NUTS concentrates mass at the dominant component ($D = 3.0\;\umsmm$ in normal
tissue) whereas MAP (ridge regression) smooths across adjacent bins due to the
$L_2$ penalty.
Posterior coefficient of variation (CV) from NUTS is annotated above the
Normal PZ panel: only $D = 0.25$ (CV = 0.20) and $D = 3.0$ (CV = 0.32) are
well identified; intermediate components have CV $> 0.74$.
}
\label{fig:spectra}
\end{figure}

% --- Fig 2: ROC Curves With Individual Components ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_roc.pdf}
\capt[Classification Performance For Tumor Detection And Gleason Grading.]{%
ROC curves from leave-one-out cross-validation ($C = 1.0$).
Thin colored lines: individual spectral components used as single-feature
classifiers.
Thick lines: ADC (blue, raw rank), MAP 8-feature LR (green), NUTS 8-feature
LR (orange dashed).
Bootstrap 95\% confidence intervals shown in brackets.
(Left)~PZ tumor detection: ADC 0.951 [0.90--0.99].
(Center)~TZ tumor detection: ADC 0.979 [0.94--1.00].
(Right)~GGG classification ($n = 29$): all methods limited by sample size.
$D = 0.25$ (restricted diffusion) is the strongest individual component across
all tasks.
}
\label{fig:roc}
\end{figure}

% --- Fig 3: ADC vs Spectral Discriminant ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_adc_discriminant.pdf}
\capt[ADC Is A Near-Optimal Spectral Discriminant.]{%
ADC versus the spectral discriminant score (logistic regression linear predictor
from eight MAP spectral fractions) across ROIs.
(Left)~Peripheral zone ($n = 81$, $r = -0.975$).
(Right)~Transition zone ($n = 68$, $r = -0.982$).
Tumor ROIs (red) cluster at low ADC / high discriminant; normal ROIs (blue) at
high ADC / low discriminant.
The near-perfect anti-correlation demonstrates that ADC rank-orders patients
nearly identically to the optimal spectral classifier.
}
\label{fig:adc_discriminant}
\end{figure}

% --- Fig 4: ADC Sensitivity vs LR Coefficients ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_sensitivity.pdf}
\capt[ADC Sensitivity Aligns With The Optimal Classification Direction.]{%
Normalized ADC sensitivity vector ($-\partial\text{ADC}/\partial R_j$, blue)
and logistic regression coefficients (red) for each diffusivity component.
(Left)~PZ: $r = -0.979$ (all 8), $r = -0.982$ (excluding $D = 20$).
(Right)~TZ: $r = -0.931$ (all 8), $r = -0.941$ (excluding $D = 20$).
Sensitivity evaluated at the average tumor spectrum (MAP features, $C = 1.0$).
The $D = 20$ discrepancy arises because ADC (fit to $b \leq 1000\;\text{s/mm}^2$)
is insensitive to the vascular compartment, which has fully decayed below the
fitting range.
}
\label{fig:sensitivity}
\end{figure}

% --- Fig 5: MAP vs NUTS Per-Component Scatter ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_map_nuts.pdf}
\capt[MAP And NUTS Estimates Agree For Well-Identified Components.]{%
MAP versus NUTS posterior mean fractions across 149~ROIs.
(Left)~$D = 0.25\;\umsmm$ ($r = 0.990$): near-perfect agreement.
(Center)~$D = 1.0\;\umsmm$ ($r = 0.483$): divergence reflecting ill-posedness.
(Right)~$D = 3.0\;\umsmm$ ($r = 0.825$): NUTS concentrates mass at the dominant
component whereas MAP (ridge regression) distributes mass across adjacent bins
due to the $L_2$ penalty, explaining the systematic offset from the identity line.
}
\label{fig:map_nuts}
\end{figure}

% --- Fig 6: Uncertainty Propagation ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_uncertainty.pdf}
\capt[Posterior Uncertainty Correlates With Misclassification.]{%
NUTS-based LOOCV predictions (sorted by predicted probability) with error bars
representing spectral feature uncertainty (mean posterior standard deviation
across components).
Circles: normal (blue); squares: tumor (red); crosses: misclassified.
Misclassified samples show 1.2--1.3$\times$ higher spectral uncertainty
than correctly classified samples, suggesting that posterior uncertainty
provides an intrinsic quality metric for classification confidence.
(Left)~PZ, (Center)~TZ, (Right)~GGG.
}
\label{fig:uncertainty}
\end{figure}

% --- Fig 7: Direction Comparison ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_directions.png}
\capt[Direction Independence Of Spectral Estimates.]{%
Per-direction MAP spectra (colored bars: 3 encoding directions) versus the
trace-averaged spectrum (black line) for 8 representative prostate voxels
(patient 8640, sorted by descending signal intensity).
The three encoding directions yield consistent spectral profiles, validating
trace averaging for spectral decomposition.
Direction variability is lowest for restricted diffusion
($D = 0.25\;\umsmm$, mean CV = 27\%) and highest for the vascular proxy
($D = 20\;\umsmm$, mean CV = 80\%).
}
\label{fig:directions}
\end{figure}
