% --- Fig 1: Mean spectra by tissue type ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_spectra.pdf}
\capt[Diffusivity spectra by tissue type and anatomical zone.]{%
Box plots of the eight NUTS posterior mean spectral fractions for normal (blue)
and tumor (red) ROIs in the peripheral zone (top) and transition zone (bottom).
Tumor tissue shows elevated restricted diffusion ($D = 0.25\;\umsmm$) and
reduced free water ($D = 3.0\;\umsmm$) compared to normal tissue.

normal tissue, while tumor spectra are more uniformly distributed across
lower diffusivities.

% --- Fig 2: ROC curves (final, verified) ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_roc.pdf}
\capt[Classification performance for tumor detection and Gleason grading.]{%
ROC curves from leave-one-out cross-validation at $C = 1.0$.
(Left)~PZ tumor detection ($n = 81$): ADC raw rank AUC = 0.951, MAP 8-feature
LR = 0.919, NUTS 8-feature LR = 0.926.
(Center)~TZ tumor detection ($n = 68$): ADC = 0.979, MAP = 0.945, NUTS = 0.923.
(Right)~Gleason grade classification ($n = 29$): ADC = 0.811, MAP = 0.733, NUTS = 0.772.
ADC consistently matches or exceeds multi-component classifiers for tumor detection,
consistent with its near-optimal alignment with the spectral discriminant.
ADC (raw) uses the feature value directly without classifier training;
ADC (LR) wraps ADC in logistic regression for the LR-comparable score.
}
\label{fig:roc}
\end{figure}

% --- Fig 3: ADC vs Spectral Discriminant (HEADLINE) ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_adc_discriminant.pdf}
\capt[ADC is a near-optimal spectral discriminant.]{%
ADC versus the spectral discriminant score (linear predictor from logistic regression
on eight MAP spectral fractions) across ROIs.
(Left)~Peripheral zone ($n = 81$, $r = -0.975$).
(Right)~Transition zone ($n = 68$, $r = -0.982$).
The near-perfect anti-correlation demonstrates that ADC rank-orders patients nearly
identically to the optimal eight-component spectral classifier.
Tumor ROIs (red) cluster at low ADC / high discriminant; normal ROIs (blue) at
high ADC / low discriminant.
}
\label{fig:adc_discriminant}
\end{figure}

% --- Fig 4: ADC Sensitivity vs LR Coefficients ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_sensitivity.pdf}
\capt[ADC sensitivity aligns with the optimal classification direction.]{%
Normalized ADC sensitivity vector ($-\partial\text{ADC}/\partial R_j$, blue) and
logistic regression coefficients (red) for each diffusivity component.
(Left)~PZ: $r = -0.979$ across all 8 components; $r = -0.982$ excluding
$D = 20.0\;\umsmm$.
(Right)~TZ: $r = -0.931$ (all 8); $r = -0.941$ (excluding $D = 20.0$).
Sensitivity evaluated at the average tumor spectrum using MAP features.
The D = 20.0 component shows a discrepancy because ADC (fit to $b \leq 1000\;\text{s/mm}^2$)
is insensitive to the vascular/CSF compartment, which decays completely below the
fitting range.
Excluding this component, the alignment improves further, confirming that ADC weights
the resolvable spectral compartments in near-optimal proportion.
}
\label{fig:sensitivity}
\end{figure}

% --- Fig 5: Per-Component Identifiability ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_identifiability.pdf}
\capt[Per-component identifiability from the NUTS posterior.]{%
(Left)~Posterior coefficient of variation (CV = std~/~mean) averaged across 149~ROIs.
Green indicates well-constrained components, red indicates poorly constrained.
Only restricted diffusion ($D = 0.25\;\umsmm$, CV = 0.20) and free water
($D = 3.0$ and $20.0\;\umsmm$, CV = 0.32--0.36) are reliably estimated.
Intermediate components ($D = 0.50$--$1.00\;\umsmm$, CV $> 0.80$) are effectively
undetermined by the 15 b-value measurements.
(Right)~Fraction--uncertainty plane: each point is one component.
The diagonal (CV = 1.0) separates components where the mean exceeds the uncertainty
from those where it does not.
$D = 0.25$ stands out as the most identifiable component relative to its magnitude.
}
\label{fig:identifiability}
\end{figure}

% --- Fig 6: MAP vs NUTS per-component scatter ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_map_nuts.pdf}
\capt[MAP and NUTS agree for well-identified components.]{%
MAP versus NUTS posterior mean fractions across 149~ROIs.
(Left)~$D = 0.25\;\umsmm$ ($r = 0.990$): near-perfect agreement, confirming both
methods robustly estimate the restricted diffusion fraction.
(Center)~$D = 1.0\;\umsmm$ ($r = 0.483$): substantial divergence, reflecting
the ill-posedness of intermediate components.
(Right)~$D = 3.0\;\umsmm$ ($r = 0.825$): moderate agreement with NUTS showing
more dynamic range.
Dashed line: identity.
Tumor (red) and normal (blue) ROIs separate along the restricted diffusion axis.
}
\label{fig:map_nuts}
\end{figure}

% --- Fig 9: Direction comparison ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_directions.png}
\capt[Direction independence of spectral estimates.]{%
Per-direction MAP spectra (colored bars: direction 1 red, 2 green, 3 blue) versus
the trace-averaged spectrum (black line) for 8 representative prostate voxels
(patient 8640, sorted by descending b=0 signal).
The three encoding directions yield qualitatively consistent spectral profiles,
with the trace average tracking the individual directions.
Direction variability was lowest for the clinically relevant restricted diffusion
component ($D = 0.25\;\umsmm$, mean CV = 27\%) and highest for the vascular
proxy ($D = 20.0\;\umsmm$, mean CV = 80\%).
This validates the use of direction-averaged (trace) signals for spectral
decomposition.
}
\label{fig:directions}
\end{figure}
