% --- Fig 4: ADC vs Spectral Discriminant (HEADLINE) ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_adc_discriminant.pdf}
\capt[ADC is a near-optimal spectral discriminant.]{%
Scatter plots of ADC versus the spectral discriminant score (linear predictor from
logistic regression on eight MAP spectral fractions) across ROIs.
(Left)~Peripheral zone ($n = 81$): $r = -0.975$, $p < 10^{-52}$.
(Right)~Transition zone ($n = 68$): $r = -0.982$, $p < 10^{-48}$.
The near-perfect anti-correlation demonstrates that ADC rank-orders patients
nearly identically to the optimal eight-component spectral classifier.
Tumor ROIs (red) cluster at low ADC / high discriminant values; normal ROIs (blue)
cluster at high ADC / low discriminant values.
}
\label{fig:adc_discriminant}
\end{figure}

% --- Fig 5: ADC Sensitivity vs LR Coefficients ---
\begin{figure}[p]

\capt[ADC sensitivity aligns with the optimal classification direction.]{%
Comparison of the ADC sensitivity vector ($-\partial\text{ADC}/\partial R_j$,

(Left)~Peripheral zone: $r = -0.979$.
(Right)~Transition zone: $r = -0.931$.
Sensitivity evaluated at the average tumor spectrum using MAP features ($C = 1.0$).
Both vectors assign the largest weights to restricted diffusion ($D = 0.25\;\umsmm$)

tumor--normal contrast.

% --- Fig 6: Per-Component Identifiability ---
\begin{figure}[p]
\centering
\includegraphics[width=0.8\textwidth]{figures/fig_identifiability.pdf}
\capt[Per-component identifiability from the NUTS posterior.]{%
Mean coefficient of variation (CV = posterior std / posterior mean) across 149~ROIs
for each diffusivity component.
Only restricted diffusion ($D = 0.25\;\umsmm$, CV = 0.20), free water
($D = 3.0\;\umsmm$, CV = 0.32), and the perfusion proxy ($D = 20.0\;\umsmm$,
CV = 0.36) fall below the identifiability threshold (dashed line, CV = 0.40).
Intermediate components ($D = 0.50$--$1.00\;\umsmm$, CV $> 0.80$) are
effectively undetermined by the 15 b-value measurements at clinical noise levels.
}
\label{fig:identifiability}
\end{figure}

% --- Fig 7: NUTS Trace Plot (simulation validation) ---
\begin{figure}[p]
\centering
\includegraphics[width=\textwidth]{figures/fig_trace_simulation.pdf}
\capt[NUTS convergence on simulated data.]{%
Posterior density estimates (left) and trace plots (right) for all eight spectral
components and the noise parameter $\sigma$ from a synthetic signal decay
(SNR = 500, ridge prior $\lambda = 0.1$).
Four independent chains of 2000~draws each show excellent mixing
(max $\hat{R} = 1.003$).
Well-identified components (e.g., $D = 0.25$, $D = 3.0$) show tight posterior
densities, while intermediate components ($D = 0.50$--$1.00$) show broad,
overlapping posteriors consistent with ill-posedness.
The noise parameter $\sigma$ is accurately recovered (posterior median near true value).
