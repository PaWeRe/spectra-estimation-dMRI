\section{Introduction}

\subsection{Problem Formulation}

Following the Stejskal-Tanner model \cite{TBD}, the observed signal decay $\mathbf{s} \in \mathbb{R}^m$ is modeled as a Laplace transform over a discrete set of non-negative diffusivity components $\mathbf{D} \in \mathbb{R}^n$, with corresponding relative fractions $\mathbf{R} \in \mathbb{R}_{\geq 0}^n$:

\begin{equation}
    \mathbf{s} = \mathbf{U} \mathbf{R}, \quad 
    \mathbf{U}_{ji} = e^{-b_j D_i}
\end{equation}

This is a linear but ill-posed inverse Laplace problem \cite{prange2009quantifying,berman2013laplace}. Assuming additive Gaussian noise with variance $\sigma^2$, the likelihood is:

\begin{equation}
    p(\mathbf{s} | \mathbf{R}) = \mathcal{N}(\mathbf{s}; \mathbf{U} \mathbf{R}, \sigma^2 \mathbf{I})
\end{equation}

To obtain full uncertainty over $\mathbf{R}$, we place a prior (e.g., uniform or truncated Gaussian) and compute the posterior via Bayes' rule. For inference, we use Gibbs sampling to draw from the truncated multivariate normal posterior \( p(\mathbf{R}|\mathbf{s}) \), enabling estimation of means, variances, and credible intervals over $\mathbf{R}$.

\subsection{Predictive Modeling with Marginalization}

To link reconstructed spectra to clinical endpoints (e.g. Gleason scores), we aim to model:

\begin{equation}
    p(\text{Gleason} \mid \mathbf{s}) = \int p(\text{Gleason} \mid \mathbf{R}) \, p(\mathbf{R} \mid \mathbf{s}) \, d\mathbf{R}
\end{equation}

We approximate this marginalization using Monte Carlo samples \( \{ \mathbf{R}^{(i)} \}_{i=1}^N \) from the posterior:

\begin{equation}
    p(\text{Gleason} \mid \mathbf{s}) \approx \frac{1}{N} \sum_{i=1}^N p(\text{Gleason} \mid \mathbf{R}^{(i)})
\end{equation}

This allows full propagation of uncertainty from signal to spectrum to tumor prediction. Each sample \( \mathbf{R}^{(i)} \) is passed through a trained classifier to yield a predictive distribution, which is then aggregated. The predictive entropy quantifies classification uncertainty and can inform downstream decision-making, such as biopsy recommendation.

This Bayesian formulation enables principled uncertainty quantification and allows comparing our method to standard approaches like ADC, which lack uncertainty estimates. In clinical workflows, this could help assess whether predictions are trustworthy and when further acquisition or biopsy is needed (????).

$A_i$ and corresponding diffusivities $D_i$.
We require $A_i \ge 0$ and $\sum_i A_i = 1$.

The predicted signal will then be
\begin{equation}
 s = E_{p(d|A,D)} [ \rho e^{-bd}] = \int \sum_{i=1}^n A_i \delta(d-D_i) \rho e^{-bd} dd = \rho \sum_{i=1}^n A_i e^{-bD_i}

The only constraint on $R$ is that $R_i \geq 0$.

\subsubsection{Posterior on $R$}

\begin{equation}
  p(R|s,D,b,K) \propto N(R; 0, K) H(R) \prod_{j=1}^m N(s_j;  R^T U^{(j)}, \sigma^2)
\end{equation}  

\subsubsection{Assuming  all $R_i$ are non-negative}
\begin{equation}
  \ln p(R|s,D,b,K) = \frac{-1}{2} R^T K^{-1} R + \sum_{j=1}^m  \frac{-1}{2\sigma^2} [s_j -  R^T U^{(j)}]^2   + C
\end{equation}  
where $C$ is a constant that does not depend on $R$. Then,
\begin{equation}
  \ln p(R|s,D,b,K) = \frac{-1}{2} R^T K^{-1} R + \sum_{j=1}^m \frac{-1}{2\sigma^2} [-2s_jR^T U^{(j)} + R^T U^{(j)}U^{(j)T}R] + C
\end{equation}  

As the log of $ p(R|s,D,b,K)$ is quadratic in $R$, it follows that
$ p(R|s,D,b,K)$ is proportional to a (multivariate) Gaussian density.

Form of multivariate Gaussian density with mean $M$ and covariance $\Sigma$:
\begin{equation}
   N(R; M, \Sigma) \propto \exp(\frac{-1}{2}[(R-M)^T \Sigma^{-1} (R-M)])
\end{equation}
or
\begin{equation}
 \ln N(R; M, \Sigma) = \frac{-1}{2}[(R-M)^T \Sigma^{-1} (R-M)] + C
\end{equation}
or
\begin{equation}
 \ln N(R; M, \Sigma) = \frac{-1}{2}[R^T \Sigma^{-1} R -2 R^T \Sigma^{-1} M ] + C
\end{equation}  

Identification of quadratic and linear terms in $R$, and solving
yields:

\begin{equation}
 \Sigma^{-1}  = K^{-1} + \frac{1}{\sigma^2} \sum_{j=1}^m  U^{(j)}U^{(j)T} \quad \mbox{and} \quad  M = \frac{1}{\sigma^2} \Sigma[  \sum_{j=1}^m s_j U^{(j)}]

\subsection{Solving for Mode}
Suppose we want the most probable diffusivity distribution,
$R^*$
\begin{equation}
   R^* = \arg \max_R \ln p(R|s,D,b,K) : R \geq 0
\end{equation} 
where $R \geq 0$ means $\forall_i R_i \geq 0$
. Then,
\begin{equation}
    R^* = \arg \max_R \frac{-1}{2}[R^T \Sigma^{-1} R -2 R^T \Sigma^{-1} M ]: R \geq 0
\end{equation}  
This can be written:
\begin{equation}
   R^* = \arg \min_R \frac{1}{2} R^T P R + Q^T R : G R \leq 0
\end{equation}
where
\begin{equation}
 P = \Sigma^{-1} = K^{-1} + \frac{1}{\sigma^2} \sum_{j=1}^m  U^{(j)}U^{(j)T} \quad \mbox{and} \quad Q = - \Sigma^{-1} M
\end{equation}
and $G = -I$. This is a standard form of a  `quadratic programming' problem.

\paragraph{Quadratic Form of Log Univariate Normal}

By definition,
\begin{equation}
  N(X_1; \mu, \sigma^2) \propto \exp(- \frac{1}{2} \frac{(X_1 - \mu)^2}{\sigma^2})
\end{equation}
or,
\begin{equation}
  N(X_1; \mu, \sigma^2) \propto \exp(- \frac{1}{2}( a'X_1^2 + b'X_1))
\end{equation}
where
\begin{equation}
  a' = \frac{1}{\sigma^2}  \qquad\mbox{and}\qquad b' = \frac{-2\mu}{\sigma^2}
\end{equation}  

Setting $a=a'$ and $b=b'$, we may solve for $\mu$ and $\sigma^2$ as follows,
\begin{equation}
  \sigma^2 = \frac{1}{\Sigma^{-1}_{11}} \qquad\mbox{and}\qquad \mu = M_1 + \frac{\Sigma^{-T}_{1/1}}{\Sigma^{-1}_{11}} (M_{/1} - X_{/1})
\end{equation}  

Thus, 
\begin{equation}
  X_1 | X_{/1} \sim N(\mu, \sigma^2)
\end{equation}  

Or, more generally,
\begin{equation}
  X_i | X_{/i} \sim N(\mu, \sigma^2)
\end{equation}  
where
\begin{equation}
  \sigma^2 = \frac{1}{\Sigma^{-1}_{ii}} \qquad\mbox{and}\qquad \mu = M_i + \frac{\Sigma^{-T}_{i/i}}{\Sigma^{-1}_{ii}} (M_{/i} - X_{/i})
\end{equation}

\section{Mean Field Variational Bayes}
We want to approximate the difficult to compute posterior distribution \(P(R|s,U)\) with an "easier" variational distribution, assuming mean field.
\subsection{Mathematical Derivation}
As distance measure to find the closest approximate distribution to the true posterior, we use KL-Divergence. 
As family of variational distributions we use Gamma distributions under mean field assumption (m refers to the number of diffusivities we are considering, in our case m=13):
\[Q_{\phi}(R) = \prod_{i=1}^{m} q_{\phi_{i}}(R_{i}),\] with \[q_{\phi_{i}}(R_{i}) = \frac{\beta^{\alpha}}{\Gamma(\alpha)}R_{i}^{(\alpha-1)}e^{-\beta R_{i}}\]
\\
The posterior distribution to be approximated follows a truncated multivariate normal distribution:
\[P(R|s,U) \propto \mathcal{N}(R | M,\Sigma)\] with \[R \in (0, \infty]\]
\\
In the following we are using the reverse KL-Divergence for it's "mode seeking" property (forward KL would be "mean seeking"):
\\
\begin{align*}
    KL(Q_{\phi}(R) || P(R|s,U)) 
                                & = \mathbb{E}_{R \sim Q_{\phi}}[log(Q_{\phi}(R)) - log(P(R|s,U))] \\
                                & = \mathbb{E}_{R \sim Q_{\phi}}[log(Q_{\phi}(R))] - \mathbb{E}_{R \sim Q_{\phi}}[log(P(R|s,U))] \\
\end{align*}
\\
\\
We can now look at both expectations separately and try to simplify as far as possible. First, as the first expectation is equal to the negative entropy of a Gamma distribution, we can write (\(\Gamma(\alpha_{i})\) referring to the Gamma function and \(\psi(\alpha_{i})\) referring to the Digamma function):
\\
\begin{align*}
    \mathbb{E}_{R \sim Q_{\phi}}[log(Q_{\phi}(R))] 
                                & = {\mathbb{E}_{R \sim Q_{\phi}}[\sum_{i=1}^{m}log(q_{\phi_{i}}(R_{i})]} \\
                                & = {\sum_{i=1}^{m}\mathbb{E}_{R_{i} \sim q_{\phi_{i}}}[log(q_{\phi_{i}}(R_{i})]} \\
                                & = -{\sum_{i=1}^{m}H(R_{i})} \\
                                & = -[{\sum_{i=1}^{m}\alpha_{i}-ln\beta_{i}+ln\Gamma(\alpha_{i})+(1-\alpha_{i})\psi(\alpha_{i})}] \\
\end{align*}
\\
\\
Looking at the second expectation we can simplify as follows (see references for some of the some of the reformulations):
\\
\begin{align*}
    \mathbb{E}_{R \sim Q_{\phi}}[log(P(R|s,U))] 
                                & = {\mathbb{E}_{R \sim Q_{\phi}}[-\frac{1}{2}[(R-M)^{T}\Sigma^{-1}(R-M)^{T}]+C]} \\
                                & = -\frac{1}{2}{\mathbb{E}_{R \sim Q_{\phi}}[R^{T}\Sigma^{-1}R-2R^{T}\Sigma^{-1}M+C]} \\
                                & \propto -\frac{1}{2}{\mathbb{E}_{R \sim Q_{\phi}}[R^{T}\Sigma^{-1}R]} +{\mathbb{E}_{R \sim Q_{\phi}}[R^{T}\Sigma^{-1}M]} \\

\end{align*}

\end{align*}

\subsection{Stejskal Tanner Expression}

\begin{equation}
  s(u) = s_0 \exp(-b d(u))
\end{equation}

$s(u)$: signal strength

$s_0$: signal strength without diffusion gradient pair

$u$: unit direction vector

$b \doteq \gamma^2 g^2 \Delta^2(\tau - 1/3 \Delta)$

$d(u)$: diffusitivity profile along $u$

$\gamma$: gyromagnetic ratio

$g$: gradient strength

$\Delta$: pulse duration

$\tau$: delay between pusles

\subsection{Isotropic Case}

Suppose $d$ is not a function of direction, then
\begin{equation}
  s = s_0 \exp(-b d)
\end{equation}  

Suppose we have an experiment with several $b$ values:
$b_{i = 1}^n$, define corresponding signals $s_i$.
Also have IID Gaussian noise $\epsilon_i \sim N(0, \sigma^2)$.


\begin{equation}
  s_i = s_0 \exp(-b_i d) + \epsilon_i
\end{equation}  

Then
\begin{equation}
p(s_i|d) = N(s_i - s_0 \exp(-b d); 0, \sigma^2)
\end{equation}  


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Define
\begin{equation}
  s_0 = \rho
\end{equation}

\begin{equation}
  s = \rho e^{-bd}
\end{equation}  
This is consistent with the S.T. model:  $s = s_0 e^{-bd}$.  Here $\rho$ is an
overall sensitivity value.

\subsection{Multi-Compartment Model}

Now suppose $d$ is a random variable with a parametric distribution
\begin{equation}
 p(d|A,D) \doteq \sum_{i=1}^n A_i \delta(d-D_i)
\end{equation}  
The parameters are the mixing fractions
$A_i$ and corresponding diffusivities $D_i$.
We require $A_i \ge 0$ and $\sum_i A_i = 1$.

The predicted signal will then be
\begin{equation}
 s = E_{p(d|A,D)} [ \rho e^{-bd}] = \int \sum_{i=1}^n A_i \delta(d-D_i) \rho e^{-bd} dd = \rho \sum_{i=1}^n A_i e^{-bD_i}
\end{equation}  

Let $R_i \doteq \rho A_i$. Then $\sum_i R_i = \rho$, and $A_i = \frac{R_i}{\sum_i R_i}$.
The only constraint on $R$ is that $R_i \geq 0$.

\subsection{Observation Model}

Now suppose there are $m$ signals $s_j$ corresponding to b-values $b_j$.
Then,
\begin{equation}
 s_j = \sum_{i=1}^n R_i e^{-b_jD_i}
\end{equation}  
or
\begin{equation}
 s_j = R^T U^{(j)}
\end{equation}  
where $U^{(j)} = [e^{-b_jD_1}, ..., e^{-b_jD_n}]^T$ are the
decays corresponding to b-value $b_j$.



Under the IID Gaussian noise model our observation model is,
\begin{equation}
  p(s|R,D,b) = \prod_{j=1}^m N(s_j;  R^T U^{(j)}, \sigma^2)
\end{equation}


\subsection{Truncated Multivarite Normal Prior on $R$}

Suppose we have a truncated multi-variate normal prior on $R$
with zero mean and covariance $K$,
\begin{equation}
  p(R) \propto N(R; 0,K) H(R)
\end{equation}
where $H(R) = 1$ if $\forall_i \; R_i \geq 0$, otherwise
$H(R) = 0$.

\subsubsection{Posterior on $R$}

Using Bayes' rule to obtain the posterior on $R$,


\begin{equation}
  p(R|s,D,b,K) \propto N(R; 0, K) H(R) \prod_{j=1}^m N(s_j;  R^T U^{(j)}, \sigma^2)
\end{equation}  

\subsubsection{Assuming  all $R_i$ are non-negative}
\begin{equation}
  \ln p(R|s,D,b,K) = \frac{-1}{2} R^T K^{-1} R + \sum_{j=1}^m  \frac{-1}{2\sigma^2} [s_j -  R^T U^{(j)}]^2   + C
\end{equation}  
where $C$ is a constant that does not depend on $R$. Then,
\begin{equation}
  \ln p(R|s,D,b,K) = \frac{-1}{2} R^T K^{-1} R + \sum_{j=1}^m \frac{-1}{2\sigma^2} [-2s_jR^T U^{(j)} + R^T U^{(j)}U^{(j)T}R] + C
\end{equation}  

As the log of $ p(R|s,D,b,K)$ is quadratic in $R$, it follows that
$ p(R|s,D,b,K)$ is proportional to a (multivariate) Gaussian density.

Form of multivariate Gaussian density with mean $M$ and covariance $\Sigma$:
\begin{equation}
   N(R; M, \Sigma) \propto \exp(\frac{-1}{2}[(R-M)^T \Sigma^{-1} (R-M)])
\end{equation}
or
\begin{equation}
 \ln N(R; M, \Sigma) = \frac{-1}{2}[(R-M)^T \Sigma^{-1} (R-M)] + C
\end{equation}
or
\begin{equation}
 \ln N(R; M, \Sigma) = \frac{-1}{2}[R^T \Sigma^{-1} R -2 R^T \Sigma^{-1} M ] + C
\end{equation}  

Identification of quadratic and linear terms in $R$, and solving
yields:

\begin{equation}
 \Sigma^{-1}  = K^{-1} + \frac{1}{\sigma^2} \sum_{j=1}^m  U^{(j)}U^{(j)T} \quad \mbox{and} \quad  M = \frac{1}{\sigma^2} \Sigma[  \sum_{j=1}^m s_j U^{(j)}]
\end{equation}  

Depending on $K$, there may not be problems with singularity
of $\Sigma^{-1}$
when  the number of measurments $m$ is less than $n$, the number of
diffusivities.

\subsubsection{Stop $R_i$ Assumption}

So we have,
\begin{equation}
 p(R|s,D,b,K) \propto N(R; M, \Sigma)
\end{equation}
in the case that all $R_i$ are non-negative,
otherwise $p(R|s,D,b,K) = 0$.

\subsection{Solving for Mode}
Suppose we want the most probable diffusivity distribution,
$R^*$
\begin{equation}
   R^* = \arg \max_R \ln p(R|s,D,b,K) : R \geq 0
\end{equation} 
where $R \geq 0$ means $\forall_i R_i \geq 0$
. Then,
\begin{equation}
    R^* = \arg \max_R \frac{-1}{2}[R^T \Sigma^{-1} R -2 R^T \Sigma^{-1} M ]: R \geq 0
\end{equation}  
This can be written:
\begin{equation}
   R^* = \arg \min_R \frac{1}{2} R^T P R + Q^T R : G R \leq 0
\end{equation}
where
\begin{equation}
 P = \Sigma^{-1} = K^{-1} + \frac{1}{\sigma^2} \sum_{j=1}^m  U^{(j)}U^{(j)T} \quad \mbox{and} \quad Q = - \Sigma^{-1} M
\end{equation}
and $G = -I$. This is a standard form of a  `quadratic programming' problem.


\subsection{Gibbs Sampling $R$}

Suppose
\begin{equation}
  p(R|M, \Sigma) \propto  N(R; M, \Sigma)
\end{equation}

Initial computation:
\begin{itemize}
  \item Set $R$ to some value, $R^*$ is natural,
  \item Initialize collection of $R$ samples to be empty
  \item For all $i$ calculate:
    \begin{itemize}
      \item  $\Sigma_{/ii}$ as in appendix
      \item  $\Sigma_{i/i}$ as in appendix
      \item  $\sigma^2_i$ as in appendix
    \end{itemize}  
\end{itemize}


Do multiple Gibbs sweeps:
\begin{itemize}
  \item For all $i$:
  \begin{itemize}
    \item Calculate $\mu_i$ \mbox{as in appendix (for} R \mbox{not} X)
    \item $R_i \leftarrow$ sample from truncated normal $N(\mu_i, \sigma^2_i) : R_i \geq 0$
  \end{itemize}
  \item Append $R$ to collection of samples
\end{itemize}

Initial computation:
\begin{itemize}
  \item Set $R$ to some value, $R^*$ is natural,
  \item Initialize collection of $R$ samples to be empty
  \item For all $i$ calculate:
    \begin{itemize}
      \item  $\Sigma^{-1}_{ii}$ as in appendix
      \item  $\Sigma^{-1}_{i/i}$ as in appendix
    \end{itemize}  
\end{itemize}

Do multiple Gibbs sweeps:
\begin{itemize}
  \item For all $i$:
  \begin{itemize}
    \item Calculate $\mu_i$ \mbox{as in appendix (for} R \mbox{not} X)
    \item $\sigma^2_i$ as in appendix
    \item $R_i \leftarrow$ sample from truncated normal $N(\mu_i, \sigma^2_i) : R_i \geq 0$
  \end{itemize}
  \item Append $R$ to collection of samples
\end{itemize}

\subsection{Conditional on $\sigma^2$}

From Eq. 19 and 
assuming no priors on $R$,
\begin{equation}
  p(R|s,D,b,\sigma^2) \propto N(R; M, \Sigma)
\end{equation}  
where
\begin{equation}
 \Sigma^{-1}  = \frac{1}{\sigma^2} \sum_{j=1}^m  U^{(j)}U^{(j)T} \quad \mbox{and} \quad  M = \frac{1}{\sigma^2} \Sigma[  \sum_{j=1}^m s_j U^{(j)}]
\end{equation}  

Let $W^{-1} \doteq \sum_{j=1}^m U^{(j)} U^{(j)T}$, then
$\Sigma^{-1} \doteq \frac{1}{\sigma^2} W^{-1}$. Note that $M$ does
not depend on $\sigma^2$.


Then
\begin{equation}
  p(R|s,D,b,\sigma^2) \propto N(R; M, \sigma^2 W)
\end{equation}

Assume uninformative (Jeffreys) prior on $\sigma^2$, {\bf GET REFERENCE}
\begin{equation}
p(\sigma^2) \propto \frac{1}{\sigma^2}  
\end{equation}

Then
\begin{equation}
  p(\sigma^2|R, s, D, b) \propto   N(R; M, \sigma^2 W) \;  p(\sigma^2)
\end{equation}
or
\begin{equation}
  p(\sigma^2|R, s, D, b) \propto   |\sigma^2 W|^{-\frac{1}{2}} \exp(- \frac{1}{2} (R-M)^T \frac{1}{\sigma^2} W^{-1} (R-M)) \; \frac{1}{\sigma^2}  
\end{equation}
or
\begin{equation}
   p(\sigma^2|R, s, D, b) \propto  \sigma^{-k - 2} |W|^{-\frac{1}{2}} \exp(- \frac{1}{2} (R-M)^T  \frac{1}{\sigma^2} W^{-1} (R-M))
\end{equation}
or
\begin{equation}
   p(\sigma^2|R, s, D, b) \propto  \sigma^{-k - 2}  \exp(- \frac{c}{\sigma^2})
\end{equation}
where $c = \frac{(R-M)^T W^{-1} (R-M)}{2} =  \sigma^2 \frac{(R-M)^T \Sigma^{-1} (R-M)}{2}$

Then
\begin{equation}
   p(\sigma^2|R, s, D, b) \propto   (\frac{1}{\sigma^2})^{\alpha+1} e^{\frac{-\beta}{\sigma^2}}
\end{equation}
where $\beta = c$ and $\alpha = \frac{k+2}{2} -1$.

Thus {\bf CHECK ME}
\begin{equation}
      \sigma^2|R, s, D, b \sim \mbox{INVGAMMA}(\frac{k+2}{2} -1, c)
\end{equation}
where  
\begin{equation}
\mbox{INVGAMMA}(x; \alpha, \beta)  \propto x^{-(\alpha + 1)} e^{-\frac{\beta}{x}}
\end{equation}

\subsection{Multivariate Normal Scalar Conditionals}
These are needed by the Gibbs sampler.  Suppose
\begin{equation}
  X \sim N(M, \Sigma)
\end{equation}

Let
\begin{equation}
  X_{/i} \doteq \mbox{`$X$ with element at row $i$ removed'}
\end{equation}
and similarly for $M_{/i}$.

The distribution on a single element of $X$, conditioned on the others is:
\begin{equation}
   X_i | X_{/i} \sim N(\mu_i, \sigma^2_i)
\end{equation}
where
\begin{equation}
  \mu_i = M_i + \Sigma^T_{i/i} \Sigma^{-1}_{/ii} (X_{/i} - M_{/i})
\end{equation}
or
\begin{equation}
  \mu_i = M_i + S^T_i (X_{/i} - M_{/i})
\end{equation}
where
\begin{equation}
  S^T_i \doteq \Sigma^T_{i/i} \Sigma^{-1}_{/ii}
\end{equation}
.  Note that $S_i$ is a column vector.

The variance is:
\begin{equation}
  \sigma^2_i = \Sigma_{ii} -  \Sigma^T_{i/i} \Sigma^{-1}_{/ii} \Sigma_{i/i} = \Sigma_{ii} -  S_i^T \Sigma_{i/i}
\end{equation}
,
\begin{equation}
  \Sigma_{/ii} \doteq \mbox{`$\Sigma$ with row $i$ and column $i$ deleted'}
\end{equation}
and
\begin{equation}
  \Sigma_{i/i} \doteq \mbox{`i'th column of $\Sigma$ with item at row $i$ deleted'}
\end{equation}

\subsubsection{Using only Inverse Covariance}

\paragraph{Quadratic form of Log Conditional}

\begin{equation}
  N(X_1;X_{/1}, M, \Sigma) \doteq \frac{N(X; M, \Sigma)}{\int N(X; M, \Sigma) dX_1}  \propto  N(X; M,\Sigma) \propto \exp(- \frac{1}{2}Q)
\end{equation}
where
\begin{equation}
  Q \doteq (X - M)^T \Sigma^{-1} (X - M)
\end{equation}
Then,
\begin{equation}
  Q = X^T \Sigma^{-1} X -2 X^T \Sigma^{-1} M + C  
\end{equation}  
where $C$ is not a function of $X$. Then
\begin{equation}
Q = 
\begin{bmatrix}
  X_1 \\
  X_{/1} 
\end{bmatrix}^T
\begin{bmatrix}
  \Sigma^{-1}_{11}   & \Sigma^{-T}_{1/1}\\
  \Sigma^{-1}_{1/1}   & \Sigma^{-1}_{/11}
\end{bmatrix}  
\begin{bmatrix}
  X_1 \\
  X_{/1} 
\end{bmatrix}
-2
\begin{bmatrix}
  X_1 \\
  X_{/1} 
\end{bmatrix}^T
\begin{bmatrix}
  \Sigma^{-1}_{11}   & \Sigma^{-T}_{1/1}\\
  \Sigma^{-1}_{1/1}   & \Sigma^{-1}_{/11}
\end{bmatrix}  
\begin{bmatrix}
  M_1 \\
  M_{/1} 
\end{bmatrix}
  + C
\end{equation}  

Or,
\begin{equation}
  Q = 
\begin{bmatrix}
  X_1 \\
  X_{/1} 
\end{bmatrix}^T
\begin{bmatrix}
  \Sigma^{-1}_{11} X_1 + \Sigma^{-T}_{1/1} X_{/1}\\
  \Sigma^{-1}_{1/1} X_1   + \Sigma^{-1}_{/11} X_{/1}
\end{bmatrix}  
-2
\begin{bmatrix}
  X_1 \\
  X_{/1} 
\end{bmatrix}^T
\begin{bmatrix}
  \Sigma^{-1}_{11} M_1 + \Sigma^{-T}_{1/1} M_{/1}\\
  \Sigma^{-1}_{1/1} M_1   + \Sigma^{-1}_{/11} M_{/1}
\end{bmatrix}  
+ C
\end{equation}  

Or, (and dropping terms constant in $X_1$),
\begin{multline}
  Q = X_1 \Sigma^{-1}_{11} X_1 + 2 X_1 \Sigma^{-T}_{1/1} X_{/1} 
  -2( X_1 \Sigma^{-1}_{11} M_1 + X_1 \Sigma^{-T}_{1/1} M_{/1}) + C
\end{multline}  

Or,
\begin{equation}
  Q = X_1^2 \Sigma^{-1}_{11} + 2X_1[\Sigma^{-T}_{1/1}(X_{/1} - M_{/1}) - \Sigma^{-1}_{11}M_1] + C
\end{equation}  

Or,
\begin{equation}
  Q = aX_1^2 + bX_1 + c
\end{equation}  
where
\begin{equation}
  a = \Sigma^{-1}_{11}  \qquad \mbox{and} \qquad b = 2[\Sigma^{-T}_{1/1}(X_{/1} - M_{/1}) - \Sigma^{-1}_{11}M_1]
\end{equation}  

\paragraph{Quadratic Form of Log Univariate Normal}

By definition,
\begin{equation}
  N(X_1; \mu, \sigma^2) \propto \exp(- \frac{1}{2} \frac{(X_1 - \mu)^2}{\sigma^2})
\end{equation}
or,
\begin{equation}
  N(X_1; \mu, \sigma^2) \propto \exp(- \frac{1}{2}( a'X_1^2 + b'X_1))
\end{equation}
where
\begin{equation}
  a' = \frac{1}{\sigma^2}  \qquad\mbox{and}\qquad b' = \frac{-2\mu}{\sigma^2}
\end{equation}  

Setting $a=a'$ and $b=b'$, we may solve for $\mu$ and $\sigma^2$ as follows,
\begin{equation}
  \sigma^2 = \frac{1}{\Sigma^{-1}_{11}} \qquad\mbox{and}\qquad \mu = M_1 + \frac{\Sigma^{-T}_{1/1}}{\Sigma^{-1}_{11}} (M_{/1} - X_{/1})
\end{equation}  

Thus, 
\begin{equation}
  X_1 | X_{/1} \sim N(\mu, \sigma^2)
\end{equation}  

Or, more generally,
\begin{equation}
  X_i | X_{/i} \sim N(\mu, \sigma^2)
\end{equation}  
where
\begin{equation}
  \sigma^2 = \frac{1}{\Sigma^{-1}_{ii}} \qquad\mbox{and}\qquad \mu = M_i + \frac{\Sigma^{-T}_{i/i}}{\Sigma^{-1}_{ii}} (M_{/i} - X_{/i})
\end{equation}
